        let arr2 = [1, 3, 5, 7];

        arr2.forEach(function (item, index, array) {
          console.log(item, index, array);
          return item;
        });

        const callback = (item, index) => {
          if (index % 2 === 0) return; // 若 index 除以 2 的餘數完全等於 0 -> 跳出迴圈
          console.log(item, index); // 不等於零的才會顯示
        }
        [0, 1, 2, 3, 4, 5, 6, 7].forEach(callback)

    JS 中變數的儲存與記憶體空間

    
        let num; // 這個階段，是物件，還沒建記憶體空間
num = 1; // 這個階段，runtime 知道這個變數是個數字，就會給他 8 byte 的記憶體
num = "Hello"; // runtime 在背後要把原本這個 8 清掉，然後開一個 2 byte 用來存放長度資訊，動態的根據長度給多個 8 byte 的記憶體空間
    所以說為什麼要在宣告時給初始值，就是為了預先做好一些事情；不要一直更換型別，是為了不要浪費效能 
    
data : {
  products: [],
  num: 0,
}

this.products = [];
this.num = 1;
